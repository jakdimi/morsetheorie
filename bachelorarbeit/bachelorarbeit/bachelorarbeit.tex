\begin{theorem}[Existenz von Morse-Funktionen]
    Sei $M \subseteq \R^n$ eine Untermannigfaltigkeit. Dann ist für fast alle
    $p \in \R^n$ die Funktion
    \begin{align*}
        f: M &\to \R \\
        x &\mapsto \lVert x - p \rVert^2
    \end{align*}
    eine Morse-Funktion.
\end{theorem}
